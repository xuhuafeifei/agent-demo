## 📡 SSE 事件类型

| Event 类型 | 用途 | 前端处理 |
|-----------|------|---------|
| `streamStart` | 流式开始 | `startStreaming()` |
| `agent_message_chunk` | AI 回复分块 | `appendStreamChunk()` |
| `agent_thought_chunk` | 思考内容分块 | `appendThinkingChunk()` |
| `tool_call` | 新工具调用 | `addToolCall()` + `breakAssistantSegment()` |
| `tool_call_update` | 工具状态更新 | `updateToolCall()` |
| `assistant_break` | 显式断开 assistant | `breakAssistantSegment()` |
| `streamEnd` | 流式结束 | `endStreaming()` |

## 🧪 后端集成示例

### Node.js Express

```javascript
import express from 'express';

const app = express();

app.post('/api/chat', async (req, res) => {
  const { message } = req.body;
  
  // 设置 SSE 响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  
  const baseTimestamp = Date.now();
  
  // 1. streamStart
  sendSSE(res, 'streamStart', { timestamp: baseTimestamp });
  
  // 2. Thinking
  sendSSE(res, 'agent_thought_chunk', {
    content: '让我分析一下...',
    timestamp: baseTimestamp
  });
  
  // 3. ToolCall
  const toolcallTimestamp = baseTimestamp + 1000;
  sendSSE(res, 'tool_call', {
    toolCallId: 'tc_001',
    toolName: 'read_file',
    title: '正在读取文件',
    content: '/path/to/file.txt',
    status: 'running',
    timestamp: toolcallTimestamp
  });
  
  // 4. ToolCall 更新
  sendSSE(res, 'tool_call_update', {
    toolCallId: 'tc_001',
    status: 'completed',
    detail: '完成 (120ms)',
    timestamp: toolcallTimestamp + 120
  });
  
  // 5. Assistant 继续（ToolCall 后）
  const assistantTimestamp = toolcallTimestamp + 2000;
  sendSSE(res, 'agent_message_chunk', {
    content: '我已经读取了文件...',
    timestamp: assistantTimestamp
  });
  
  // 6. streamEnd
  sendSSE(res, 'streamEnd');
  
  res.end();
});

function sendSSE(res, type, data) {
  res.write(`event: ${type}\n`);
  res.write(`data: ${JSON.stringify(data)}\n\n`);
}

app.listen(3000);
```

```javascript
export const useChatStore = create((set, get) => {
  // 索引追踪（类似 useRef）
  let streamingMessageIndex = null;
  let thinkingMessageIndex = null;

  return {
    // 开始流式：创建 assistant 占位消息
    startStreaming: (timestamp) => {
      const assistantIndex = state.messages.length;
      streamingMessageIndex = assistantIndex;
      // 创建空消息
    },

    // 追加 chunk：根据索引追加或创建新消息
    appendStreamChunk: (chunk, timestamp) => {
      if (streamingMessageIndex === null) {
        // 创建新消息
      } else {
        // 追加到当前消息
      }
    },

    // Thinking：独立索引追踪
    appendThinkingChunk: (chunk) => {
      if (thinkingMessageIndex === null) {
        // 创建 thinking 消息（timestamp = assistantTs - 1）
      } else {
        // 追加到 thinking 消息
      }
    },

    // ToolCall：自动断开 assistant 流
    addToolCall: (toolCall) => {
      toolCalls.push(toolCall);
      breakAssistantSegment();  // ← 关键！
    },

    // 获取所有消息并排序
    getAllMessages: () => {
      return [...messages, ...toolCalls].sort(
        (a, b) => a.timestamp - b.timestamp
      );
    },
  };
});
```

### App.jsx - 消息渲染

```javascript
function MessageList({ allMessages, isStreaming }) {
  // 自动滚动优化
  const userHasScrolledRef = useRef(false);
  
  useEffect(() => {
    // 只在用户未手动滚动时自动滚动
    if (!userHasScrolledRef.current && isStreaming) {
      el.scrollTop = el.scrollHeight;
    }
  }, [allMessages, isStreaming]);

  // 监听用户滚动
  useEffect(() => {
    const handleScroll = () => {
      const isNearBottom = el.scrollHeight - (el.scrollTop + el.clientHeight) < 120;
      userHasScrolledRef.current = !isNearBottom;
    };
    el.addEventListener('scroll', handleScroll);
  }, []);

  // 按类型渲染消息
  return allMessages.map((item) => {
    if (item.type === 'tool_call') return <ToolCallCard ... />;
    if (item.data.role === 'user') return <UserMessage ... />;
    if (item.data.role === 'thinking') return <ThinkingMessage ... />;
    if (item.data.role === 'assistant') return <AssistantMessage ... />;
  });
}
```

## 📝 注意事项

1. **后端必须返回 timestamp**：所有 SSE 事件应包含 `timestamp` 字段
2. **ToolCall 自动断开**：`addToolCall()` 会自动调用 `breakAssistantSegment()`
3. **Thinking 时间戳**：自动设置为 `assistantTimestamp - 1ms`
4. **滚动优化**：用户手动滚动后停止自动滚动
