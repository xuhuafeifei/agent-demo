### 场景 3：Thinking → ToolCall → Assistant（完整流程）

**后端发送**：
```javascript
// 1. 开始流式
sendSSE('streamStart', { timestamp: T0 });

// 2. Thinking
sendSSE('agent_thought_chunk', { content: '让我查一下这个问题...' });

// 3. ToolCall
sendSSE('tool_call', {
  toolCallId: 'tc_001',
  toolName: 'search',
  title: '正在搜索',
  content: '搜索关键词',
  status: 'running',
  timestamp: T1
});

// 4. ToolCall 更新
sendSSE('tool_call_update', {
  toolCallId: 'tc_001',
  status: 'completed',
  detail: '完成 (120ms)',
  timestamp: T2
});

// 5. Assistant 继续（ToolCall 后）
sendSSE('agent_message_chunk', {
  content: '根据搜索结果...',
  timestamp: T3
});

// 6. 结束
sendSSE('streamEnd');
```

**预期结果**：
```
[Thinking] 让我查一下这个问题...
[ToolCall] 正在搜索 ✓ 完成 (120ms)
[Assistant] 根据搜索结果...
```

## 后端集成示例

### Node.js Express 示例

```javascript
import express from 'express';

const app = express();

app.post('/api/chat', async (req, res) => {
  const { message } = req.body;
  
  // 设置 SSE 响应头
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  
  const baseTimestamp = Date.now();
  
  // 1. streamStart
  sendSSE(res, 'streamStart', { timestamp: baseTimestamp });
  
  // 2. Thinking
  sendSSE(res, 'agent_thought_chunk', {
    content: '让我分析一下你的问题...',
    timestamp: baseTimestamp
  });
  
  // 3. 模拟思考延迟
  await sleep(500);
  
  // 4. ToolCall
  const toolcallTimestamp = baseTimestamp + 1000;
  sendSSE(res, 'tool_call', {
    toolCallId: 'tc_001',
    toolName: 'read_file',
    title: '正在读取文件',
    content: '/path/to/file.txt',
    status: 'running',
    timestamp: toolcallTimestamp
  });
  
  // 5. 模拟工具执行
  await sleep(1000);
  
  // 6. ToolCall 更新
  sendSSE(res, 'tool_call_update', {
    toolCallId: 'tc_001',
    status: 'completed',
    detail: '完成 (120ms)',
    timestamp: toolcallTimestamp + 120
  });
  
  // 7. Assistant 继续
  const assistantTimestamp = toolcallTimestamp + 2000;
  sendSSE(res, 'agent_message_chunk', {
    content: '我已经读取了文件内容，',
    timestamp: assistantTimestamp
  });
  
  sendSSE(res, 'agent_message_chunk', {
    content: '文件内容如下...',
    timestamp: assistantTimestamp + 100
  });
  
  // 8. streamEnd
  sendSSE(res, 'streamEnd', { timestamp: Date.now() });
  
  res.end();
});

function sendSSE(res, type, data) {
  res.write(`event: ${type}\n`);
  res.write(`data: ${JSON.stringify(data)}\n\n`);
}

function sleep(ms) {
  return new Promise(resolve => setTimeout(resolve, ms));
}

app.listen(3000);
```

## 验证要点

### ✅ 必需要点

1. **Thinking 在 Assistant 前面显示**
   - Thinking 时间戳 = `assistantTimestamp - 1`
   
2. **ToolCall 在 Thinking 和 Assistant 之间显示**
   - ToolCall 时间戳 > Thinking 时间戳
   - ToolCall 时间戳 < Assistant 时间戳
   
3. **ToolCall 后 Assistant 继续输出**
   - ToolCall 自动调用 `breakAssistantSegment()`
   - 下一个 `agent_message_chunk` 创建新消息

4. **消息按时间戳排序**
   - 使用 `getAllMessages()` 获取排序后的消息

## 常见问题

### Q1: Thinking 没有显示？

**检查**：
- 后端是否发送了 `agent_thought_chunk` 事件
- 事件格式是否正确：`{ content: '...' }`

### Q2: ToolCall 后 Assistant 没有分段？

**检查**：
- ToolCall 是否调用了 `addToolCall()`
- `addToolCall()` 后是否自动调用了 `breakAssistantSegment()`

### Q3: 消息顺序不对？

**检查**：
- 所有消息是否有 `timestamp` 字段
- 是否使用了 `getAllMessages()` 进行排序
