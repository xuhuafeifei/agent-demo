```typescript
// 当前：简单的事件监听
// 改进：Hook 系统

type HookFn = (context: AgentContext) => Promise<void>

class AgentHookManager {
  private hooks: {
    beforePrompt: HookFn[]
    afterPrompt: HookFn[]
    beforeToolExecution: HookFn[]
    afterToolExecution: HookFn[]
  } = {
    beforePrompt: [],
    afterPrompt: [],
    beforeToolExecution: [],
    afterToolExecution: []
  }

  register(phase: keyof typeof this.hooks, hook: HookFn) {
    this.hooks[phase].push(hook)
  }

  async execute(phase: keyof typeof this.hooks, context: AgentContext) {
    for (const hook of this.hooks[phase]) {
      await hook(context)
    }
  }
}

// 使用示例
const hookManager = new AgentHookManager()

// 记忆提取 Hook
hookManager.register('afterPrompt', async (context) => {
  if (shouldExtractMemory(context)) {
    await extractMemories(context)
  }
})

// 日志记录 Hook
hookManager.register('afterToolExecution', async (context) => {
  logToolUsage(context)
})
```

- ✅ 更好的扩展性

**agent-demo 可以改进**：
```typescript
// 当前：三层配置
// 改进：更结构化的配置管理

interface ConfigLayer {
  priority: number
  source: string
  config: Partial<AgentConfig>
}

class ConfigManager {
  private layers: ConfigLayer[] = []

  addLayer(layer: ConfigLayer) {
    this.layers.push(layer)
    this.layers.sort((a, b) => b.priority - a.priority)
  }

  getConfig(): AgentConfig {
    return this.layers.reduce(
      (merged, layer) => ({ ...merged, ...layer.config }),
      {} as AgentConfig
    )
  }
}

// 使用示例

// 环境变量（最高优先度）

  priority: 100,
  source: 'environment',

// 项目配置

  priority: 50,
  source: 'project',
  config: readProjectConfig()
})

// 全局配置

  priority: 10,
  source: 'global',
  config: readGlobalConfig()
```

## 不适合学习的设计

### 1. 四层安全防护

**原因**：
- ❌ 过度复杂，不适合轻量级应用
- ❌ LLM 分类器增加成本和延迟
- ❌ 沙箱隔离对生活助手不必要

**建议**：
```typescript
// 简化版：简单的权限检查
function checkToolPermission(toolName: string, input: unknown): boolean {
  // 基础的危险操作检查
  const dangerousPatterns = [
    { tool: 'bash', pattern: /rm -rf/ },
    { tool: 'bash', pattern: /rm -r/ },
    { tool: 'bash', pattern: /dd/ }
  ]

  for (const { tool, pattern } of dangerousPatterns) {
    if (toolName === tool && pattern.test(JSON.stringify(input))) {
      return false
    }
  }

  return true
}
```

### 2. 双层记忆系统

**原因**：
- ❌ Session Memory 对轻量应用过于复杂
- ❌ Forked agent 增加系统复杂度
- ❌ 记忆压缩对生活助手不必要

**建议**：
```typescript
// 简化版：单一记忆系统
class SimpleMemoryManager {
  private memories: Memory[] = []

  addMemory(memory: Memory) {
    this.memories.push(memory)
    // 定期清理旧记忆
    this.cleanup()
  }

  search(query: string): Memory[] {
    return this.memories.filter(m =>
      m.content.toLowerCase().includes(query.toLowerCase())
    )
  }

  private cleanup() {
    const maxAge = 30 * 24 * 60 * 60 * 1000 // 30天
    const now = Date.now()
    this.memories = this.memories.filter(
      m => now - m.timestamp < maxAge
    )
  }
}
```

### 3. 30+ 专用工具

**原因**：
- ❌ 对生活助手来说过于复杂
- ❌ 维护成本高
- ❌ 大部分工具不适用

**建议**：
